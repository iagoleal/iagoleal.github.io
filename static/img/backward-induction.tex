\documentclass{standalone}

\def\pgfsysdriver{pgfsys-dvisvgm.def}
\usepackage{tikz,pgffor,ifthen}

\usetikzlibrary{animations, positioning, arrows, arrows.meta}

\begin{document}

\colorlet{smoothgray}{gray!80}

\begin{tikzpicture}[
  node distance = 3cm,
  state/.style    = { circle, minimum width = 0.7cm, fill = gray!20, very thick },
  terminal/.style = {rectangle, fill = black, scale = 2},
  action/.style   = {
    opacity = 0.8,
    thin,
    smoothgray,
    -{Kite[length=4pt,width=2.5pt,inset=1.5pt]},
  },
  animate/pulse/.style = {
    myself: = {
      begin on = {begin, of = "state"},
      scope = {
        time = #1,
        :color = "orange",
        :line width = "0.4pt",
      },
      0.8s = {
        :color = "orange!80!red",
        :line width = "0.8pt",
        entry,
      },
      20s = {
        :color = "orange!80!red",
        :line width = "0.8pt",
        entry
      },
    },
    myself: = {
      begin on = {end, of next = "apocalypse"}, % Undo the blue marks when the animation ends
      id = 2,
      scope = {
        attribute = color,
        0s = "current value",
        0.5s = "smoothgray",
        freeze,
        entry,
      },
      scope = {
        attribute = line width,
        0s = "current value",
        0.5s = "0.4pt",
        freeze,
        entry,
      },
      entry,
    },
  },
]

  % Useful constants
  \def\nStages{4}
  \def\nStates{3}
  \pgfmathsetseed{5671243}

  \node :fill = { 0s         = "current value",
                  10ms later = "current value",
                  name = "state",
                  begin = 0.5s,
                  begin on = {end, of next = "apocalypse", delay = 1s},
                  }
        [terminal] (T) at (3*\nStages-1.5, 1.5) {};

  \foreach \t in {\nStages,...,1} {
    \foreach \s in {\nStates,...,1} {
      % Grid of states
      \node :fill = { 0s = { exit control = {1}{0}, "gray!20" },
                      0.2s     = "orange",
                      0.8s     = "orange",
                      1s = { entry control = {0}{1}, "gray!20" },
                      begin on = {end, of = "state", delay = -0.2s},
                      name = "state",
                      }
        [state, draw = gray!40, outer sep = 2] (q\t\s) at (3*\t-3, 1.5*\s-1.5) {};

      % Arraws going out of each state
      \ifthenelse{\t < \nStages}{
        \pgfmathtruncatemacro{\next}{\t + 1}
        \pgfmathrandominteger{\bestaction}{1}{\nStates}

        \foreach \ny in {1,...,\nStates} {
            \ifnum \ny=\bestaction
              \draw [action, animate ={pulse={\ny*100ms}}] (q\t\s) -- (q\next\ny);
            \else
              \draw :color = {
                      \ny*100ms = {exit control = {1}{0}, "orange"},
                      0.8s      = "smoothgray",
                      begin on = {begin, of = "state"},
                      }
                  [action] (q\t\s) -- (q\next\ny);
            \fi
        }
      }
      {
        \draw [animate={pulse={0s}}, action] (q\t\s) -- (T);
      }
    }
  }

  \node :visible = {0s = "false", 3s = "false", begin on = {end, of = "state"}, name = "apocalypse"} {};
\end{tikzpicture}

\end{document}
